- 无传递依赖 -> 3NF

- 已满足3NF, 无需分解
- performances表增长最快, 建议按赛季分区

## 实验七: 视图管理

在phpMyAdmin -> SQL, 每条执行后截图:

-- (1) 创建视图
CREATE VIEW v_teams AS SELECT * FROM teams;

-- (2) 查询视图
SELECT * FROM v_teams;

-- (3) 创建视图: 2022年后比赛
CREATE VIEW v_matches_recent AS
SELECT match_id, match_number, match_date FROM matches WHERE season_id >= 4;

-- (4) 查询视图
SELECT * FROM v_matches_recent;

-- (5) 创建视图: 球员表现汇总
CREATE VIEW v_player_stats AS
SELECT p.player_id, p.player_name,
SUM(pe.goals) AS total_goals,
SUM(pe.assists) AS total_assists,
AVG(pe.rating) AS avg_rating
FROM players p JOIN performances pe ON p.player_id = pe.player_id
GROUP BY p.player_id;

-- (6) 查询视图
SELECT * FROM v_player_stats;

-- (7) 查看视图结构
DESC v_player_stats;

-- (8) 查看视图创建信息
SHOW CREATE VIEW v_player_stats;

-- (9) 替换视图
CREATE OR REPLACE VIEW v_player_stats AS
SELECT p.player_id, p.player_name, SUM(pe.goals) AS total_goals
FROM players p JOIN performances pe ON p.player_id = pe.player_id
GROUP BY p.player_id HAVING SUM(pe.goals) > 0;

-- (10) 删除视图
DROP VIEW v_matches_recent;

## 实验八: 索引管理

在phpMyAdmin -> SQL, 每条执行后截图:

-- (1) 创建索引
CREATE INDEX idx_player_name ON players(player_name);

-- (2) 分析查询
EXPLAIN SELECT * FROM players WHERE player_name = 'Erling Haaland';

-- (3) 复合唯一索引
ALTER TABLE players ADD UNIQUE INDEX idx_name_pos (player_name, position);

-- (4) 比较不同查询条件
EXPLAIN SELECT * FROM players WHERE player_name = 'Erling Haaland';
EXPLAIN SELECT * FROM players WHERE position = 'FWD';

-- (5) 全文索引
ALTER TABLE players ADD COLUMN player_info TEXT;
UPDATE players SET player_info = player_name;
CREATE FULLTEXT INDEX idx_info ON players(player_info);

-- (6) 查看索引
SHOW INDEX FROM players;

-- (7) 删除索引
ALTER TABLE players DROP INDEX idx_name_pos;

-- (8) 清理
ALTER TABLE players DROP INDEX idx_info;
ALTER TABLE players DROP COLUMN player_info;
